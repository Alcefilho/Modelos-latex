    footnotes within tables

    \usepackage{setspace}
\usepackage{threeparttable}
\begin{table}
\begin{threeparttable} 
\begin{tabular}{|l|c|r|} 
\hline 
A & B & C \\
1 & 2 & 3 \tnote{1} \\
\end{tabular} 
\begin{tablenotes}\footnotesize 
\item[1] Prognose 2003 
\end{tablenotes}
\end{threeparttable}
\end{table}
    The command \tnote{...} "replace \footnote{...}"
